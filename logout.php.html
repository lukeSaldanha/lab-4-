/* 
   Logout Page 
   This script logs the user out by destroying their session and redirects them to the login page. 
*/

// Start or resume the current session
session_start();

// Remove all session variables
session_unset();

// Destroy the current session
session_destroy();

// Start a new session to set a logout confirmation message
session_start();
$_SESSION['message'] = "You have successfully logged out.";

// Redirect the user to the login page
header("Location: login.php");
exit();
?>
